{%- macro format_list(element_list) -%}
    {%- for elem in element_list -%}
        \VAR{ elem }
        {%- if loop.revindex == 2 %} and {% elif not loop.last %}, {% endif -%}
    {%- endfor -%}
{%- endmacro -%}
